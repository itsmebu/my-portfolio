void loadInventory() {
    FILE *file = fopen(FILENAME, "rb");
    if (file == NULL) {
        itemCount = 0;
        return;
    }
    
    // Get file size
    fseek(file, 0, SEEK_END);
    long fileSize = ftell(file);
    fseek(file, 0, SEEK_SET);
    
    if (fileSize == 0) {
        fclose(file);
        itemCount = 0;
        return;
    }
    
    // Calculate number of items
    int numItems = fileSize / sizeof(HardwareItem);
    
    if (numItems > MAX_ITEMS) {
        numItems = MAX_ITEMS;
    }
    
    itemCount = fread(inventory, sizeof(HardwareItem), numItems, file);
    fclose(file);
}

void saveInventory() {
    if (itemCount == 0) {
        printf("No items to save!\n");
        return;
    }
    
    FILE *file = fopen(FILENAME, "wb");
    if (file == NULL) {
        printf("Error: Cannot create or open inventory file for writing!\n");
        return;
    }
    
    size_t itemsWritten = fwrite(inventory, sizeof(HardwareItem), itemCount, file);
    fclose(file);
    
    if (itemsWritten == itemCount) {
        printf("Inventory saved successfully! %d items saved.\n", itemCount);
    } else {
        printf("Error: Only %zu out of %d items were saved!\n", itemsWritten, itemCount);
    }
}

void generateNextId(char newId[]) {
    if (itemCount == 0) {
        strcpy(newId, "01");
        return;
    }
    
    // Find the highest ID
    int maxId = 0;
    for (int i = 0; i < itemCount; i++) {
        int currentId = atoi(inventory[i].id);
        if (currentId > maxId) {
            maxId = currentId;
        }
    }
    
    // Generate next ID
    int nextId = maxId + 1;
    if (nextId < 10) {
        sprintf(newId, "0%d", nextId);
    } else if (nextId < 100) {
        sprintf(newId, "%d", nextId);
    } else {
        sprintf(newId, "99");
    }
}

void processSale() {
    if (itemCount == 0) {
        printf("No items in inventory!\n");
        return;
    }
    
    SaleItem cart[50];
    int cartCount = 0;
    int continueShopping = 1;
    
    printf("\n=== PROCESS SALE ===\n");
    
    while (continueShopping && cartCount < 50) {
        char id[5];
        int quantity;
        
        printf("\nEnter Item ID (0 to finish, -1 to cancel sale): ");
        scanf("%s", id);
        
        if (strcmp(id, "0") == 0) {
            break;
        }
        if (strcmp(id, "-1") == 0) {
            printf("Sale cancelled.\n");
            return;
        }
        
        int index = findItemById(id);
        if (index == -1) {
            printf("Item not found!\n");
            continue;
        }
        
        if (inventory[index].quantity == 0) {
            printf("Item is out of stock!\n");
            continue;
        }
        
        printf("Item: %s, Available: %d, Price: ₱%.2f\n", 
               inventory[index].name, inventory[index].quantity, inventory[index].price);
        printf("Enter quantity: ");
        scanf("%d", &quantity);
        
        if (quantity <= 0) {
            printf("Invalid quantity!\n");
            continue;
        }
        
        if (quantity > inventory[index].quantity) {
            printf("Insufficient stock! Only %d available.\n", inventory[index].quantity);
            continue;
        }
        
        // Add to cart
        strcpy(cart[cartCount].item_id, id);
        strcpy(cart[cartCount].item_name, inventory[index].name);
        cart[cartCount].quantity = quantity;
        cart[cartCount].price = inventory[index].price;
        cart[cartCount].subtotal = quantity * inventory[index].price;
        
        // Update inventory
        inventory[index].quantity -= quantity;
        
        cartCount++;
        printf("Item added to cart!\n");
        
        if (cartCount < 50) {
            printf("Add more items? (1=Yes, 0=No): ");
            scanf("%d", &continueShopping);
        }
    }
    
    if (cartCount > 0) {
        float discount = 0;
        printf("\nEnter discount percentage (0 for no discount): ");
        scanf("%f", &discount);
        
        generateBill(cart, cartCount, discount, 1); // Save order automatically
        printf("Sale completed successfully! Inventory updated.\n");
        
        // Auto-save after sale
        saveInventory();
    } else {
        printf("No items in cart. Sale cancelled.\n");
    }
}

void generateBill(SaleItem cart[], int cartCount, float discount, int saveOrder) {
    printf("\n");
    printf("============================================\n");
    printf("           MCM HARDWARE - INVOICE\n");
    printf("============================================\n");
    printf("%-20s %-8s %-12s %-12s\n", "Item", "Qty", "Price", "Subtotal");
    printf("------------------------------------------------------------\n");
    
    float total = 0;
    for (int i = 0; i < cartCount; i++) {
        printf("%-20s %-8d ₱%-11.2f ₱%-11.2f\n", 
               cart[i].item_name, 
               cart[i].quantity, 
               cart[i].price, 
               cart[i].subtotal);
        total += cart[i].subtotal;
    }
    
    printf("------------------------------------------------------------\n");
    printf("Subtotal: ₱%.2f\n", total);
    
    if (discount > 0) {
        float discountAmount = total * (discount / 100);
        printf("Discount (%.1f%%): -₱%.2f\n", discount, discountAmount);
        total -= discountAmount;
    }
    
    // Add tax (example: 12% VAT)
    float tax = total * 0.12;
    printf("VAT (12%%): ₱%.2f\n", tax);
    total += tax;
    
    printf("============================================\n");
    printf("TOTAL: ₱%.2f\n", total);
    printf("============================================\n");
    printf("Thank you for your purchase!\n");
    
    // Save order if requested
    if (saveOrder && orderCount < MAX_ORDERS) {
        Order newOrder;
        newOrder.order_id = orderCount + 1;
        
        // Get current date and time
        time_t t = time(NULL);
        struct tm tm = *localtime(&t);
        sprintf(newOrder.date_time, "%04d-%02d-%02d %02d:%02d:%02d", 
                tm.tm_year + 1900, tm.tm_mon + 1, tm.tm_mday,
                tm.tm_hour, tm.tm_min, tm.tm_sec);
        
        // Copy cart items to order
        newOrder.item_count = cartCount;
        for (int i = 0; i < cartCount; i++) {
            newOrder.items[i] = cart[i];
        }
        
        newOrder.total_amount = total + discount;
        newOrder.discount = discount;
        newOrder.tax = tax;
        newOrder.final_total = total;
        
        recentOrders[orderCount] = newOrder;
        orderCount++;
        
        // Auto-save orders
        saveOrders();
    }
}
